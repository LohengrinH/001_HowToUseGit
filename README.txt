이전 - 깃 설치 및 깃헙 가입 후 Bash로 실행
-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------

1. 초기 설정

git config --global user.name "이름"
       // ""까지 입력해야 한다.
git config --global user.email "깃허브 메일주소" // 매번 물어보는 귀찮음을 피하기 위해 설정.



-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------

2. 저장 폴더 만들기 (Ex. D드라이브에 git 폴더 생성)

cd d: 		//d드라이브 이동
mkdit ./git 	// "mkdir ./폴더명" 으로 디렉토리 생성

mkdir ~/MyProject   // c 드라이브에서 로컬 디렉토리 만드는 명령어

-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------

3. 명령어 사용할 수 있는 디렉토리로 변환
git init		// 버전 관리를 위한 도구들이 모여있는 폴더 생성

-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------

4. 만든 저장 폴더로 접근해서 파일 만들기

cd d:\git 		// 아무런 디렉토리에도 접속하지 않앗을때 한번에 해당 디렉토리로 이동한다.

echo "test md" >> "$(date +%y)$(date +%m)$(date +%d)_(date +%H)$(date +%M)_"README.md		// test md 라는 내용이 들어있는 현재 시각의 README 파일을 생성한다.

 
-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------

5. 만든 파일 확인(폴더 확인) 및 Tracked(관리 대상 파일)상태 변환

git status		// 현재 폴더의 상태를 보는 것, 자료를 추가했다면 빨간색으로 뜰것

git add <파일이름> 		// 관리 대상 파일 상태로 만드는 명령어, git add * 을 입력시 폴더 전체를 가리킴

git status 		// 입력시 녹색으로 변화된것 확인

or
git add .           // 이 명령은 현재 디렉토리의 모든 화일을 추가할 수 있다.


git remote add origin https://github.com/username/myproject.git // 로컬과 원격 저장소를 연결한다.


git remote -v // 연결상태를 확인한다.





-----------------------------------------------------------------------------------------------------
-----------------------------------------------------------------------------------------------------


cd ~/myproject      // 디렉토리로 들어가서

git init            // 깃 명령어를 사용할 수 있는 디렉토리로 만든다.

git status          // 현재 상태를 훑어보고

git add 화일명.확장자  // 깃 주목 리스트에 화일을 추가하고 
or
git add .           // 이 명령은 현재 디렉토리의 모든 화일을 추가할 수 있다.

git commit -m “현재형으로 설명” // 커밋해서 스냅샷을 찍는다.


git remote add origin https://github.com/username/myproject.git // 로컬과 원격 저장소를 연결한다.

git remote -v // 연결상태를 확인한다.

git push origin master // 깃허브로 푸시한다.

git remote remove origin - 원격 삭제
